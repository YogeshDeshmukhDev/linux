ls -l file -- show permissions, owner and group of file
chmod 777 file -- give read+write+execute to owner, group and others
chmod 755 file -- owner rwx, group and others read+execute only
chmod 700 file -- owner rwx, no access for group or others
chmod 644 file -- owner read+write, group and others read only
chmod 600 file -- owner read+write only, no access for others
chmod +x file -- add execute permission
chmod -x file -- remove execute permission
chmod +r file -- add read permission
chmod -r file -- remove read permission
chmod +w file -- add write permission
chmod -w file -- remove write permission
chmod u+x file -- add execute permission for owner (user) only
chmod g+w file -- add write permission for group only
chmod o-r file -- remove read permission for others only
chmod a+x file -- add execute permission for all (owner, group, others)
chmod -R 755 dir -- apply permissions recursively to a directory
chown user file -- change owner of file
chown user:group file -- change owner and group of file
chown -R user:group dir -- change owner and group recursively
chgrp group file -- change group of file
umask -- show default permission mask for new files
umask 022 -- set default permission mask

Permission number meanings:
r = 4  (read)
w = 2  (write)
x = 1  (execute)
7 = rwx  (4+2+1)
6 = rw-  (4+2)
5 = r-x  (4+1)
4 = r--  (4)
0 = ---  (no permission)
Order is always: owner group others  e.g. chmod 764 file = owner rwx, group rw-, others r--

19. Check prime number
#!/bin/bash
read -p "Enter a number: " n
flag=0
if [ $n -le 1 ]
then
flag=1
fi
for ((i=2; i<=n/2; i++))
do
if [ $((n % i)) -eq 0 ]
then
flag=1
break
fi
done
if [ $flag -eq 0 ]
then
echo "$n is Prime"
else
echo "$n is Not Prime"
fi

20. Palindrome check (number)
#!/bin/bash
read -p "Enter a number: " n
orig=$n
rev=0
while [ $n -gt 0 ]
do
digit=$((n % 10))
rev=$((rev * 10 + digit))
n=$((n / 10))
done
if [ $rev -eq $orig ]
then
echo "Palindrome"
else
echo "Not Palindrome"
fi

21. Reverse a number
#!/bin/bash

done
echo "Reversed Number: $rev"

23. Armstrong number check
#!/bin/bash
read -p "Enter a number: " n
orig=$n
sum=0
while [ $n -gt 0 ]
do
digit=$((n % 10))
sum=$((sum + digit * digit * digit))
n=$((n / 10))
done
if [ $sum -eq $orig ]
then
echo "Armstrong Number"
else
echo "Not an Armstrong Number"
fi

31. Count vowels in a string
#!/bin/bash
read -p "Enter a string: " str
count=0
for (( i=0; i<${#str}; i++ ))
do
ch=${str:$i:1}
case $ch in
[aeiouAEIOU]) count=$((count + 1)) ;;
esac
done
echo "Number of vowels: $count"
